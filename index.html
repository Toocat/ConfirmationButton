Welcome to the jQuery confirmativeActionButton plugin GitHub Page.

Demo page

Ease of Use

$("#someDiv").confirmativeActionButton();

Custom styling

$("#someDiv").confirmativeActionButton({
        cssNormalClass: "confirmativeActionButton",
        cssWaitingClass: "confirmativeActionButtonWaitingForConfirmation",
        cssActivatedClass: "confirmativeActionButtonConfirmed"
});

API & defaults

id: methods.guid.apply(this)
caption: "Confirmative Action Button"
confirmationCaption: "Click to confirm"
waitingTime: 3000
actionHandler: null
cssNormalClass: confirmativeActionButton
cssWaitingClass: confirmativeActionButtonWaitingForConfirmation
cssActivatedClass: confirmativeActionButtonConfirmed

Self-destruction

$("#someDiv").confirmativeActionButton({id:"actionButton", actionHandler: function (e) {
        $("#actionButton").removeConfirmativeActionButton();
}});
