Just the button.

$("#someDiv").confirmativeActionButton();

Action handler with expanded waiting time.

$("#someDiv").confirmativeActionButton({id: "actionButton1",
caption: "Click me!",
waitingTime: 5000,
actionHandler: function () {
alert("Clicked!");
}
});

Custom styling.

$("#someDiv").confirmativeActionButton({id: "actionButton2",
caption: "Click me!",
waitingTime: 5000,
cssNormalClass: "confirmativeActionButtonCustom",
cssWaitingClass: "confirmativeActionButtonWaitingForConfirmationCustom",
cssActivatedClass: "confirmativeActionButtonConfirmedCustom",
actionHandler: function () {
"alert("Clicked!");
}
});

Self-destruction.

$("#someDiv").confirmativeActionButton({id: "actionButton3",
actionHandler: function (target) {
target.removeConfirmativeActionButton();
}
});

All custom.

$("#someDiv").confirmativeActionButton({id: "actionButton2",
caption: "Click me!",
confirmationCaption: "Please Click to Confirm",
waitingTime: 2000,
cssNormalClass: "confirmativeActionButtonCustom",
cssWaitingClass: "confirmativeActionButtonWaitingForConfirmationCustom",
cssActivatedClass: "confirmativeActionButtonConfirmedCustom",
actionHandler: function () {
console.log("Click confirmed.");
}
});
